import {
  autoinject,
  bindable,
  bindingMode,
  computedFrom,
  customElement,
  inlineView
} from "aurelia-framework";
import { UIInternal } from "../utils/ui-internal";
 
@autoinject()
@customElement("ui-sidebar")
@inlineView(`<template class="ui-sidebar" click.trigger="peek = true" data-peek.bind="peek" data-collapsed.bind="collapsed" data-align.bind="align">
<div class="ui-sidebar__titlebar">
  <slot name="sidebar-title"><div class="ui-sidebar__title" if.bind="label">\${label}</div></slot>
  <div class="ui-sidebar__toggler" if.bind="collapsible" click.trigger="[collapsed = !collapsed, $event.stopEvent()]"><ui-svg-icon icon.bind="toggleIcon"></ui-svg-icon></div>
</div>
<div class="ui-sidebar__body" css.bind="{width, maxWidth}">
  <slot></slot>
</div>
</template>`)
export class UISidebar {
  @bindable()
  public align: string = "start";
  @bindable()
  public label: string;
  @bindable()
  public width: string = "24rem";
  @bindable()
  public maxWidth: string = "40vw";
 
  @bindable({ defaultBindingMode: bindingMode.twoWay })
  public collapsed: boolean = false;
 
  protected peek: boolean = false;
  protected collapsible: boolean = false;
 
  private obClick;
 
  constructor(private element: Element) {
    this.collapsible = element.hasAttribute("collapsible");
 
    this.obClick = UIInternal.subscribe(UIInternal.EVT_VIEWPORT_CLICK, () => (this.peek = false));
  }
 
  protected detached(): void {
    if (this.obClick) {
      this.obClick.dispose();
    }
  }
 
  @computedFrom("collapsed", "align")
  get toggleIcon() {
    return `${this.collapsed ? "expand" : "collapse"}-${this.align}`;
  }
}
